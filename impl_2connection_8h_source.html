46 inline error_code bind_connection_executor(Connection& conn, const Executor& ex) {
47  if (get_executor(conn) != ex) {
48  decltype(auto) socket = get_socket(conn);
49  std::decay_t<decltype(socket)> s{ex.context()};
50  error_code ec;
51  s.assign(socket.native_handle(), ec);
52  if (ec) {
53  return ec;
54  }
55  socket.release();
56  socket = std::move(s);
57  conn.io_ = std::addressof(ex.context());
58  }
